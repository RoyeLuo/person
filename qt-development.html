Qt基础

Qt是一个跨平台的C++图形用户界面应用程序开发框架，支持Windows、Linux、macOS、Android、iOS等平台。

信号与槽机制

Qt的核心机制，用于对象间的通信。

// 连接信号和槽
connect(btn, &QPushButton::clicked, this, &MyWidget::onButtonClicked);
connect(spinBox, QOverload::of(&QSpinBox::valueChanged),
           label, QOverload::of(&QLabel::setNum));

// Lambda表达式（Qt5+）
connect(btn, &QPushButton::clicked, [this](){
    qDebug() << "Button clicked!";
    this->process();
});

// 信号连接信号
connect(lineEdit, &QLineEdit::returnPressed, lineEdit, &QLineEdit::clear);

// 断开连接
disconnect(btn, &QPushButton::clicked, this, QOverload<>::of(&MyWidget::onClicked));

数据库操作

// 数据库连接
QSqlDatabase db = QSqlDatabase::addDatabase("QSQLITE");
db.setDatabaseName("myapp.db");
if (!db.open()) {
    qDebug() << "数据库打开失败:" << db.lastError().text();
}

// 创建表
QSqlQuery query;
query.exec("CREATE TABLE IF NOT EXISTS users ("
           "id INTEGER PRIMARY KEY AUTOINCREMENT, "
           "name TEXT NOT NULL, "
           "email TEXT"
           ")");

// 插入数据
query.prepare("INSERT INTO users (name, email) VALUES (:name, :email)");
query.bindValue(":name", "张三");
query.bindValue(":email", "zhang@example.com");
query.exec();

// 查询数据
query.exec("SELECT * FROM users WHERE age > 20");
while (query.next()) {
    int id = query.value("id").toInt();
    QString name = query.value("name").toString();
    qDebug() << id << name;
}

// 更新数据
query.prepare("UPDATE users SET name = :name WHERE id = :id");
query.bindValue(":name", "李四");
query.bindValue(":id", 1);
query.exec();

// 删除数据
query.exec("DELETE FROM users WHERE id = 1");

// 事务
db.transaction();
// ... 执行多个操作 ...
db.commit();  // 或 db.rollback();

多线程

// 方式1：继承 QThread
class WorkerThread : public QThread
{
    Q_OBJECT
protected:
    void run() override {
        for (int i = 0; i < 100; ++i) {
            QThread::sleep(1);
            emit progress(i);      // 发出信号
        }
    }
signals:
    void progress(int value);
};

// 使用
WorkerThread *thread = new WorkerThread;
connect(thread, &WorkerThread::progress, this, [this](int v){
    progressBar->setValue(v);
});
connect(thread, &WorkerThread::finished, thread, &QObject::deleteLater);
thread->start();

// 方式2：QRunnable + QThreadPool（推荐）
class Task : public QRunnable
{
public:
    void run() override {
        // 任务代码...
        emit finished();
    }
signals:
    void finished();
};

QThreadPool::globalInstance()->start(new Task);
注意：Qt GUI操作必须在主线程执行，跨线程通信使用信号槽

// 自定义模型示例
class MyTableModel : public QAbstractTableModel
{
    Q_OBJECT
public:
    int rowCount(const QModelIndex &parent) const override { return m_data.size(); }
    int columnCount(const QModelIndex &parent) const override { return 3; }
    QVariant data(const QModelIndex &index, int role) const override {
        if (role == Qt::DisplayRole) {
            return m_data[index.row()][index.column()];
        }
        return {};
    }
    QVariant headerData(int section, Qt::Orientation orientation, int role) const override {
        if (role == Qt::DisplayRole && orientation == Qt::Horizontal) {
            return m_headers[section];
        }
        return {};
    }
private:
    QVector<QVector<QString>> m_data;
    QStringList m_headers = {"姓名", "年龄", "城市"};
};

QML与Quick

QML是一种声明式语言，用于构建现代化的UI。

// main.qml
import QtQuick 2.15
import QtQuick.Controls 2.15

ApplicationWindow {
    visible: true
    width: 800
    height: 600
    title: "Qt QML Demo"
    
    Column {
        anchors.centerIn: parent
        spacing: 10
        
        Text {
            text: "Hello, Qt!"
            font.pixelSize: 24
            color: "#41cd52"
        }
        
        Button {
            text: "Click Me"
            onClicked: {
                console.log("Button clicked!")
            }
        }
        
        TextField {
            placeholderText: "输入内容"
            onAccepted: console.log(text)
        }
    }
}

// main.cpp 加载QML
QQmlApplicationEngine engine;
engine.load(QUrl(QStringLiteral("qrc:/main.qml")));
if (engine.rootObjects().isEmpty())
    return -1;
return app.exec();
