基础配置

# 初始化与配置
git init                        # 初始化仓库
git clone https://github.com/user/repo.git   # 克隆仓库
git clone --depth 1 repo.git   # 浅克隆（最近一次提交）

# 用户配置
git config --global user.name "Your Name"
git config --global user.email "you@example.com"
git config --global core.editor vim  # 设置默认编辑器
git config --list                # 查看配置

# SSH Key配置
ssh-keygen -t ed25519 -C "you@example.com"   # 生成SSH密钥
# 将 ~/.ssh/id_ed25519.pub 内容添加到 GitHub/GitLab
git config --global url."git@github.com:".insteadOf "https://github.com/"  # HTTPS转SSH
~/.gitconfig 示例： 
[user]
    name = Your Name
    email = you@example.com
[alias]
    co = checkout
    br = branch
    ci = commit
    st = status
    lg = log --graph --pretty=format:'%Cred%h%Creset -%C(yellow)%d%Creset %s %Cgreen(%cr) %C(bold blue)<%an>%Creset' --abbrev-commit
[push]
    default = simple
[pull]
    rebase = true

日常工作流

main: A---B---C---F---G │ │ feature: └---D---E---┘ 工作区 → 暂存区 → 本地仓库 → 远程仓库
# 查看状态
git status                      # 查看工作区状态
git status -s                   # 简洁格式
git diff                        # 工作区vs暂存区差异
git diff --cached              # 暂存区vs提交差异
git diff HEAD                  # 工作区vs最新提交
git diff branch1..branch2       # 分支差异

# 添加到暂存区
git add file.txt                # 添加单个文件
git add .                       # 添加所有更改
git add -p                      # 交互式添加（部分暂存）
git add -u                      # 只添加已跟踪文件

# 提交
git commit -m "提交信息"        # 提交
git commit -am "提交信息"       # 暂存并提交（仅已跟踪文件）
git commit --amend             # 修改最后一次提交
git commit --no-verify -m "msg"  # 跳过钩子
简洁状态： ?? 未跟踪   A 新添加   M 修改   D 删除   R 重命名

分支管理

# 查看分支
git branch                      # 本地分支
git branch -a                   # 所有分支（含远程）
git branch -r                   # 远程分支
git branch -v                   # 显示最新提交

# 创建与切换
git branch feature              # 创建分支
git checkout feature            # 切换分支（旧）
git switch feature              # 切换分支（新）
git checkout -b feature         # 创建并切换
git switch -c feature          # 创建并切换（新）

# 删除
git branch -d feature           # 删除分支（已合并）
git branch -D feature           # 强制删除
git push origin --delete feature # 删除远程分支

# 重命名
git branch -m old new           # 重命名当前分支

# 合并
git merge feature               # 合并feature到当前分支
git merge --no-ff feature       # 禁用快进合并（保留分支历史）
git merge --squash feature      # 压缩合并（多条commit变一条）

# 变基（Rebase）
git rebase main                 # 变基到main（保持线性历史）
git rebase -i HEAD~3           # 交互式变基（修改最近3条）
# pick: 保留  squash: 合并  reword: 修改信息  drop: 删除

# 暂存当前工作
git stash                       # 暂存当前更改
git stash pop                   # 恢复并删除stash
git stash apply                 # 恢复（保留stash）
git stash list                  # 查看stash列表
git stash drop                  # 删除stash
merge vs rebase：
  • merge：保留完整历史，合并commit，适合公共分支
  • rebase：线性历史，适合个人分支同步上游
  • 黄金法则：不要对已推送的提交执行rebase

远程操作

# 远程仓库管理
git remote -v                   # 查看远程仓库
git remote add origin url       # 添加远程仓库
git remote rename origin github  # 重命名
git remote remove origin        # 删除
git remote set-url origin url    # 修改URL

# 推送
git push origin main            # 推送到远程
git push -u origin feature       # 推送并设置上游分支
git push --force                # 强制推送（慎用！）
git push --force-with-lease      # 安全强制推送
git push origin --delete feature # 删除远程分支

# 拉取
git pull                        # 拉取并合并（fetch + merge）
git pull --rebase               # 拉取并变基
git fetch                       # 仅获取（不合并）
git fetch --all                # 获取所有远程

# 拉取请求/合并请求
git request-pull main origin    # 生成pull请求摘要
git pull = git fetch + git merge
git pull --rebase = git fetch + git rebase

查看历史

# 日志查看
git log                          # 完整日志
git log --oneline               # 简洁一行
git log -10                     # 最近10条
git log --graph                # 图形化分支
git log --author="name"         # 按作者筛选

git log -p file.txt            # 文件变更历史
git log --all --grep="fix"     # 按提交信息搜索

# 文件历史
git log --follow file.txt       # 文件重命名历史
git blame file.txt             # 逐行追溯
git show commit:file.txt        # 查看某提交的版本

git describe --tags             # 最近标签描述

git lg

撤销与回退

# 工作区撤销
git checkout -- file.txt         # 撤销工作区更改
git restore file.txt             # 同上（新命令）
git restore .                    # 撤销所有工作区更改

# 暂存区撤销
git reset HEAD file.txt          # 取消暂存
git reset --soft HEAD~1         # 撤销提交，保留更改在暂存
git reset --mixed HEAD~1        # 撤销提交，保留更改在工作区（默认）
git reset --hard HEAD~1         # 完全撤销（危险！）

# 回退到指定版本
git revert commit_id             # 反向合并（保留历史，推荐）
git reset --hard commit_id       # 硬回退（危险）

# 恢复误删的文件
git checkout -- file.txt         # 从暂存区恢复
git reflog                       # 找到删除前的commit
git checkout abc123 -- file.txt  # 从历史恢复
三种reset模式对比：
模式HEADIndexWorking
--soft✓✓✓
--mixed（默认）✓✓✓
--hard✓✓✓

✓ 表示保留，灰色表示改变

标签管理

# 创建标签
git tag v1.0.0                    # 轻量标签
git tag -a v1.0.0 -m "版本1.0.0"  # 附注标签（推荐）
git tag -a v1.0.0 commit_id      # 给历史commit打标签

# 查看标签
git tag                          # 列出所有标签
git tag -l "v1.*"               # 搜索标签
git show v1.0.0                  # 查看标签详情

# 推送标签
git push origin v1.0.0          # 推送单个标签
git push origin --tags          # 推送所有标签

# 删除标签
git tag -d v1.0.0               # 删除本地标签
git push origin :refs/tags/v1.0.0  # 删除远程标签
git push origin --delete v1.0.0    # 同上

GitFlow工作流

main (生产) │ ├────────────●────────────●───→ v1.0 v2.0 │ │ │ release │ │ │ ▼ ▼ └────●──────●────────────● │ │ hotfix │ │ │ ▼ ▼────●────●────●────●────→ develop │ │ │ │ │ │ ┌────┴────┴────┴────┴────┴────┴────┐ │ │ ▼ ▼ ──●────●────→ feature/login ──●────●────→ feature/payment feature feature
# GitFlow 分支命名
main/master          # 主分支（生产环境）
develop              # 开发分支
feature/xxx          # 功能分支
release/v1.0         # 发布分支
hotfix/xxx           # 热修复分支

# 开始新功能
git checkout -b feature/login develop
# ... 开发完成 ...
git checkout develop
git merge --no-ff feature/login
git branch -d feature/login

# 创建发布分支
git checkout -b release/v1.0 develop
# ... 修复bug，测试 ...
git checkout main
git merge --no-ff release/v1.0
git tag -a v1.0 -m "Version 1.0"
git checkout develop
git merge --no-ff release/v1.0
git branch -d release/v1.0

# 热修复
git checkout -b hotfix/critical main
# ... 紧急修复 ...
git checkout main
git merge --no-ff hotfix/critical
git tag -a v1.0.1 -m "Hotfix 1.0.1"
git checkout develop
git merge --no-ff hotfix/critical
git branch -d hotfix/critical

子模块与子树

# 子模块操作
git submodule add https://github.com/user/lib.git libs/lib  # 添加子模块
git submodule init                  # 初始化子模块
git submodule update                # 更新子模块
git submodule update --remote libs/lib  # 更新到远程最新
git submodule foreach git pull       # 更新所有子模块

# 子树操作
git subtree add --prefix=libs https://github.com/user/lib.git main  # 添加
git subtree pull --prefix=libs https://github.com/user/lib.git main  # 拉取
git subtree push --prefix=libs https://github.com/user/lib.git main  # 推送
子模块 vs 子树：
  • 子模块：独立仓库指针，clone时需额外命令
  • 子树：合并到主仓库，clone时自动包含

高级技巧

Cherry-pick

git cherry-pick commit_id    # 拣选单个提交
git cherry-pick commit1..commit3  # 拣选范围
git cherry-pick --continue  # 解决冲突后继续
git cherry-pick --abort      # 取消拣选

Bisect 定位bug

git bisect start            # 开始二分查找
git bisect bad              # 当前版本有bug
git bisect good v1.0        # 标记好的版本
# Git自动切换，测试后标记
git bisect good/bad         # 重复直到定位
git bisect reset           # 结束

工作区清理

git clean -n               # 预览将删除的文件
git clean -f               # 删除未跟踪文件
git clean -fd              # 删除未跟踪文件和目录
git clean -fX              # 只删除忽略的文件
git clean -fx              # 删除所有未跟踪文件

Hook钩子

# .git/hooks/ 目录
pre-commit           # 提交前执行
prepare-commit-msg   # 填写提交信息前
commit-msg           # 验证提交信息
post-commit          # 提交后执行
pre-push             # 推送前执行
