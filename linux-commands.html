文件与目录操作

# 目录切换
cd /home/user              # 切换到指定目录
cd ..                     # 返回上一级目录
cd ~                      # 返回用户主目录
cd -                      # 返回上次所在目录
pwd                       # 显示当前目录

# 目录查看
ls -la                     # 列出所有文件（含隐藏）
ls -lh                     # 人性化显示文件大小
ls -lt                     # 按修改时间排序
tree                      # 树形显示目录结构

# 文件操作
cp src.txt dst.txt         # 复制文件
cp -r dir/ newdir/        # 递归复制目录
mv old.txt new.txt         # 移动/重命名
rm file.txt                # 删除文件
rm -rf dir/               # 强制删除目录
mkdir -p a/b/c            # 创建多级目录
touch file.txt             # 创建空文件或更新时间

# 文件权限
chmod 755 script.sh        # 设置权限
chmod +x script.sh        # 添加执行权限
chown user:group file      # 修改所有者
chown -R user:group dir/  # 递归修改

# 链接
ln -s target link         # 创建软链接
ln target hardlink         # 创建硬链接

权限数字对照

数字权限含义
4r--读
2-w-写
1--x执行
7rwx读写执行
6rw-读写
5r-x读执行

文本处理

# 查看文件
cat file.txt                # 查看全部内容
head -n 20 file.txt        # 查看前20行
tail -n 50 file.txt        # 查看后50行
tail -f log.txt            # 实时追踪日志
less file.txt              # 分页查看（q退出，/搜索）
wc -l file.txt             # 统计行数

# 文本搜索
grep "error" log.txt         # 搜索关键词
grep -r "pattern" dir/       # 递归搜索
grep -i "error" log.txt     # 忽略大小写
grep -n "error" log.txt      # 显示行号
grep -v "debug" log.txt      # 反向匹配
grep -E "error|warning"     # 扩展正则

# 文本处理
awk '{print $1, $3}' file.txt    # 提取列（$1第一列，$NF最后一列）
awk '/pattern/ {print $0}' file.txt   # 模式匹配
sed 's/old/new/g' file.txt     # 全局替换
sed -i 's/old/new/g' file.txt   # 直接修改文件
cut -d',' -f1,3 file.csv    # 按分隔符提取列
sort file.txt              # 排序
uniq file.txt              # 去重（需先排序）
wc -l file.txt             # 统计行数

# 管道组合
cat log.txt | grep "error" | awk '{print $1}' | sort | uniq -c | sort -rn
awk 常用变量： NF（字段数）、NR（行号）、$0（整行）、$1-$n（各字段）、FS（字段分隔符）

系统管理

# 进程管理
ps aux                       # 查看所有进程
ps -ef                       # 详细进程信息
top                          # 动态进程监控（q退出）
htop                         # 更友好的进程监控
kill PID                     # 终止进程
kill -9 PID                  # 强制终止
pkill -f nginx               # 按名称终止进程
nohup ./script.sh &          # 后台运行（忽略挂断）

# 系统监控
top                          # CPU/内存监控
free -h                       # 内存使用情况
df -h                         # 磁盘使用情况
du -sh *                      # 目录大小
uptime                        # 系统运行时间
who                           # 当前登录用户
w                             # 详细登录信息

# 系统信息
uname -a                     # 系统信息
cat /etc/os-release           # 操作系统版本
hostname                      # 主机名
lscpu                         # CPU信息
lsblk                         # 块设备（磁盘）

# 服务管理（systemd）
systemctl start nginx         # 启动服务
systemctl stop nginx          # 停止服务
systemctl restart nginx       # 重启服务
systemctl status nginx        # 查看状态
systemctl enable nginx        # 开机自启
systemctl disable nginx       # 取消自启
systemctl list-unit-files     # 列出所有服务

网络命令

# 网络配置
ifconfig                      # 查看网络接口（旧）
ip addr                        # 查看IP（新）
ip link set eth0 up            # 启用网卡
ip addr add 192.168.1.100/24 dev eth0   # 添加IP

# 网络测试
ping -c 4 google.com           # Ping测试（4次）
traceroute google.com          # 路由追踪
mtr google.com                 # 实时路由追踪

# 端口与连接
netstat -tuln                  # 监听端口
ss -tuln                      # 更快的端口查看
netstat -anp                   # 所有连接
lsof -i :8080                  # 端口占用进程

# 远程操作
ssh user@host                  # SSH连接
scp file.txt user@host:/path/  # 远程复制
rsync -avz src/ user@host:/path/  # 增量同步

# 下载
wget https://example.com/file.zip     # 下载文件
curl -O https://example.com/file.zip   # 下载
curl -X POST -d '{"key":"value"}' url   # POST请求
curl -H "Authorization: Bearer xxx" url  # 带Header
ss vs netstat：
  • ss -tuln 比 netstat -tuln 更快
  • ss -tp 显示进程信息
  • ss -s 显示统计摘要

压缩与归档

# tar 归档
tar -cvf archive.tar dir/       # 创建归档
tar -xvf archive.tar           # 解压归档
tar -cvzf archive.tar.gz dir/  # gzip压缩
tar -xvzf archive.tar.gz      # 解压gzip
tar -cvjf archive.tar.bz2 dir/ # bzip2压缩
tar -xvjf archive.tar.bz2     # 解压bzip2

# zip
zip -r archive.zip dir/        # 压缩
unzip archive.zip              # 解压
unzip -l archive.zip          # 查看内容

# 其他
gzip file.txt                  # gzip压缩（删除原文件）
gunzip file.txt.gz             # 解压
xz -z file.txt                # xz压缩（高压缩比）

用户与权限管理

# 用户管理
useradd -m -s /bin/bash username   # 创建用户
userdel -r username                # 删除用户
usermod -aG sudo username          # 加入sudo组
passwd username                     # 设置密码
id username                          # 查看用户信息

# 组管理
groupadd groupname                  # 创建组
groups username                      # 查看用户组

# sudo权限
visudo                               # 编辑sudoers（安全方式）
# username ALL=(ALL:ALL) ALL  # 添加这行

# su切换
su username                        # 切换用户
sudo command                       # 以root执行

磁盘与文件系统

# 分区与格式化
fdisk -l                           # 查看磁盘分区
parted /dev/sdb                   # 交互式分区工具
mkfs.ext4 /dev/sdb1               # 格式化ext4
mkfs.xfs /dev/sdb1                # 格式化xfs

# 挂载
mount /dev/sdb1 /mnt/data         # 挂载
umount /mnt/data                   # 卸载
mount -o remount,rw /               # 重新挂载
df -hT                              # 查看挂载和文件系统

# 逻辑卷管理（LVM）
pvcreate /dev/sdb1                 # 创建物理卷
vgcreate vg01 /dev/sdb1           # 创建卷组
lvcreate -L 100G -n lv01 vg01     # 创建逻辑卷
lvextend -L +50G /dev/vg01/lv01   # 扩展逻辑卷
resize2fs /dev/vg01/lv01          # 扩展文件系统

定时任务与日志

# crontab 定时任务
crontab -e                        # 编辑定时任务
crontab -l                        # 查看定时任务
crontab -r                        # 删除所有任务

# 格式：分 时 日 月 周 命令
# * * * * * /path/to/command
# 0 2 * * * /backup.sh          # 每天凌晨2点
# */5 * * * * /check.sh         # 每5分钟
# 0 0 1 * * /monthly.sh         # 每月1号
# 0 2 * * 0 /weekly.sh          # 每周日凌晨2点

# systemd 定时器（推荐）
# /etc/systemd/system/backup.timer
[Timer]
OnCalendar=*-*-02:00:00
Persistent=true

# 日志查看
journalctl                          # systemd日志
journalctl -u nginx.service        # 服务日志
journalctl -f                      # 实时追踪
journalctl --since "1 hour ago"  # 最近1小时

# 日志文件
tail -f /var/log/syslog           # 系统日志
tail -f /var/log/nginx/access.log # Nginx访问日志
注意：定时任务中的路径最好使用绝对路径，避免环境变量问题

查找与搜索

# 文件查找
find / -name "*.log"               # 按名称查找
find / -type f -size +100M         # 查找大文件
find / -type d -name "node_modules" # 查找目录
find / -mtime -7                  # 7天内修改的文件
find / -user www-data             # 按所有者查找
find / -perm 755                 # 按权限查找

# 快速查找（数据库索引）
locate filename                   # 需要updatedb更新数据库
which python3                     # 命令路径
whereis python3                   # 二进制、源码、man位置
type python3                       # 命令类型

# locate 示例
updatedb                           # 更新文件数据库
locate "*.conf"                  # 查找配置文件

!grep                     # 执行最近grep

alias ll='ls -la'

# 永久保存到 ~/.bashrc

环境变量

echo $PATH              # 查看
export VAR=value         # 设置
env                      # 列出所有

管道与重定向

cmd > file.txt          # 输出到文件
cmd >> file.txt         # 追加
cmd 2> err.txt          # 错误重定向
cmd &> all.txt          # 所有输出
cmd1 | cmd2             # 管道
