削峰填谷 - 限流策略

Peak Shaving & Valley Filling - Rate Limiting
核心思想：在流量高峰时限制请求，防止系统过载；在流量低谷时提前处理堆积任务。 
// 令牌桶算法实现 - 滑动窗口限流
public class TokenBucketRateLimiter {
    private readonly int capacity;        // 桶容量
    private readonly double refillRate;     // 每秒补充令牌数
    private double tokens;
    private long lastRefillTime;
    
    public TokenBucketRateLimiter(int capacity, double refillRate) {
        this.capacity = capacity;
        this.refillRate = refillRate;
        this.tokens = capacity;
        this.lastRefillTime = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
    }
    
    public bool TryAcquire() {
        Refill();
        if (tokens >= 1) {
            tokens -= 1;
            return true;
        }
        return false;
    }
    
    private void Refill() {
        long now = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
        double elapsed = (now - lastRefillTime) / 1000.0;
        tokens = Math.Min(capacity, tokens + elapsed * refillRate);
        lastRefillTime = now;
    }
}
// ASP.NET Core 限流中间件
public class RateLimitingMiddleware {
    private static readonly ConcurrentDictionary<string, DateTime> requests = new();
    
    public static async Task Invoke(HttpContext context) {
        string clientId = context.Connection.RemoteIpAddress?.ToString();
        var now = DateTime.UtcNow;
        
        // 每秒最多100个请求
        requests[GetKey(clientId, now)] = now;
        CleanOldRequests(clientId);
        
        if (requests.Count(r => GetKey(clientId, r.Value) == GetKey(clientId, now)) > 100) {
            context.Response.StatusCode = 429;
            await context.Response.WriteAsJsonAsync(new { error = "请求过于频繁" });
            return;
        }
    }
}
应用场景：API限流、秒杀系统、爬虫限制、恶意访问防护

幂等性设计

Idempotency Design
核心思想：相同的操作执行一次和执行多次的结果相同，确保重复请求不会产生副作用。 
// 1. 数据库唯一索引 + 插入前检查
public async Task<bool> CreateOrder(Order order) {
    try {
        await _context.Orders.AddAsync(order);
        await _context.SaveChangesAsync();
        return true;
    } catch (DbUpdateException) {
        // 唯一索引冲突，订单已存在
        return false;
    }
}

// 2. 基于Token的幂等
public async Task<string> GenerateIdempotencyKey() {
    return Guid.NewGuid().ToString();
}

public async Task<Result> ProcessPayment(string idempotencyKey, PaymentRequest request) {
    if (await _cache.ExistsAsync($"idempotent:{idempotencyKey}")) {
        return await _cache.GetAsync<Result>($"result:{idempotencyKey}");
    }
    Result result = await DoPayment(request);
    await _cache.SetAsync($"idempotent:{idempotencyKey}", "1", TimeSpan.FromHours(24));
    await _cache.SetAsync($"result:{idempotencyKey}", result, TimeSpan.FromHours(24));
    return result;
}
适用场景：支付扣款、订单创建、库存扣减、消息消费、HTTP重试

缓存设计策略 - 解决缓存三兄弟问题

Cache Design - Solving Cache Problems
缓存三兄弟：缓存雪崩、缓存击穿、缓存穿透 
// 1. 缓存雪崩：大量缓存同时过期 → 解决：过期时间 + 随机偏移
public class CacheService {
    public async Task<T> GetOrSetAsync<T>(string key, Func<Task<T>> factory, int baseExpireMinutes = 30) {
        T value = await _redis.GetAsync<T>(key);
        if (value != null) return value;
        
        value = await factory();
        // 添加随机偏移量，避免同时过期
        int randomMinutes = new Random().Next(1, 10);
        await _redis.SetAsync(key, value, TimeSpan.FromMinutes(baseExpireMinutes + randomMinutes));
        return value;
    }
}

// 2. 缓存击穿：热点key过期瞬间大量请求 → 解决：互斥锁/永不过期
private static readonly AsyncLock _lock = new AsyncLock();
public async Task<T> GetWithLock<T>(string key, Func<Task<T>> factory) {
    T value = await _redis.GetAsync<T>(key);
    if (value != null) return value;
    
    using (await _lock.LockAsync(key)) {
        value = await _redis.GetAsync<T>(key);
        if (value != null) return value;
        value = await factory();
        await _redis.SetAsync(key, value, TimeSpan.FromMinutes(30));
    }
    return value;
}

// 3. 缓存穿透：查询不存在的数据 → 解决：布隆过滤器/空值缓存
public async Task<User> GetUser(int id) {
    if (!_bloomFilter.MightContain(id)) {
        return null;  // 布隆过滤器判断不存在，直接返回
    }
    User user = await _redis.GetAsync<User>($"user:{id}");
    if (user != null) return user;
    
    user = await _db.Users.FindAsync(id);
    if (user == null) {
        // 缓存空值，防止穿透
        await _redis.SetAsync($"user:{id}", "NULL", TimeSpan.FromMinutes(5));
    } else {
        await _redis.SetAsync($"user:{id}", user, TimeSpan.FromMinutes(30));
    }
    return user;
}

日志规范与最佳实践

Logging Standards 
// 日志级别使用规范
_log.Trace()   // 跟踪：详细的调试信息，生产环境禁用
_log.Debug()   // 调试：开发调试使用，生产环境可关闭
_log.Info()    // 信息：正常业务操作记录
_log.Warning() // 警告：潜在问题但不影响功能
_log.Error()   // 错误：功能受损，需要关注
_log.Fatal()   // 致命：系统崩溃，无法继续

// 结构化日志 - 使用日志上下文
using (_log.BeginScope(new Dictionary<string, object> {
    ["OrderId"] = orderId,
    ["UserId"] = userId
})) {
    _log.Information("订单创建成功");
}

// 输出格式：JSON（便于日志收集分析）
{"level":"INFO","time":"2026-01-01T10:00:00Z","message":"订单创建成功","OrderId":"12345","UserId":"67890"}
日志记录原则：
  • 禁止记录敏感信息（密码、身份证、银行卡号）
  • 日志要包含上下文（用户ID、订单ID、追踪ID）
  • 异常日志必须记录堆栈信息
  • 关键业务操作需要记录入参和出参

异常处理规范

Exception Handling Standards 
// 1. 区分业务异常和技术异常
public class BusinessException : Exception {
    public string Code { get; }
    public BusinessException(string code, string message) : base(message) {
        Code = code;
    }
}

// 2. 全局异常处理
public class GlobalExceptionHandler : ExceptionHandler {
    public override async Task HandleAsync(ExceptionHandlerContext context) {
        var exception = context.Exception;
        var result = exception switch {
            BusinessException be => new ObjectResult(new { code = be.Code, message = be.Message }) { StatusCode = 400 },
            UnauthorizedAccessException => new ObjectResult(new { code = "UNAUTHORIZED", message = "未授权" }) { StatusCode = 401 },
            _ => new ObjectResult(new { code = "INTERNAL_ERROR", message = "系统错误" }) { StatusCode = 500 }
        };
        context.Result = result;
    }
}
注意事项：
  • 不要捕获所有异常后静默处理
  • 不要在异常处理中抛出新异常
  • finally块中释放资源
  • 异步方法使用await而非同步Wait()

代码审查清单

Code Review Checklist

安全检查

  • 用户输入是否做了验证和过滤
  • SQL是否使用参数化查询
  • 敏感数据是否加密存储
  • 权限校验是否完整
  • 是否防止了XSS/CSRF攻击

健壮性检查

  • 空值/边界条件是否处理
  • 异常是否被正确捕获
  • 资源是否正确释放
  • 并发问题是否考虑
  • 超时和重试机制是否完善

性能检查

  • 数据库查询是否加了索引
  • 是否存在N+1查询问题
  • 大对象是否及时释放
  • 循环中是否避免重复计算
  • 缓存策略是否合理

代码规范

  • 命名是否符合规范
  • 方法是否过长（建议<50行）
  • 是否有重复代码
  • 注释是否清晰必要
  • 是否遵循SOLID原则
